import * as fs from 'fs';
import * as path from 'path';
 
import { getMerger } from './merger';
 
interface File {
    path: string;
    description: string;
}
 
interface FileContent {
    content: string;
    description: string;
}
 
interface JsonMap {
    [p: string]: string | number | boolean | null | JsonMap | JsonMap[];
}
 
type ParsePropertyStrategy = (name: string, fullName: string, hasMore: boolean, target: any) => any;
 
export interface BuildArgs {
    configName: string;
    dir?: string;
    env?: string;
    CONFIG?: string;
    baseExt?: string;
    [p: string]: any;
}
 
export const defaultDir = '.';
export const defaultExt = 'json';
 
const readFile = (file: string): Promise<string> => {
    return new Promise<string>((resolve, reject) => {
        fs.readFile(file, { encoding: 'utf8' }, (err, data) => {
            if (err) {
                reject(err);
            } else {
                resolve(data);
            }
        });
    });
};
 
const readJson = ({path: filePath, description}: File): Promise<string> => {
    return readFile(filePath)
        .then(content => JSON.parse(content))
        .catch(() => {
            throw new Error(`${description} "${filePath}" does not exist`);
        });
};
 
const getBaseConfigFile = (args: BuildArgs): File => {
    return {
        path: path.join(args.dir || defaultDir, `${args.configName}.${args.baseExt || defaultExt}`),
        description: 'Base config file'
    };
};
 
const getEnvConfigFile = (args: BuildArgs): File => {
    return {
        path: path.join(args.dir || defaultDir, `${args.configName}.${args.env}.json`),
        description: 'Environment config file'
    };
};
 
const parseConfig = ({content, description}: FileContent): string => {
    try {
        return content ? JSON.parse(content) : {};
    } catch (err) {
        throw new Error(`${description} is invalid JSON`);
    }
};
 
const buildConfig = (values: JsonMap, includePattern: RegExp, strategy: ParsePropertyStrategy): any => {
    return Object.keys(values)
        .filter(name => includePattern.test(name))
        .reduce((config, name) => {
            const [, ...props] = name.split(':');
            let target = config;
            while (props.length) {
                target = strategy(props.shift(), name, !!props.length, target);
            }
            return config;
        }, {});
};
 
const buildConcatConfig = (values: JsonMap): any => {
    const configNamePattern = /^CONFIG(?::([^:\[\]]+(?:\[])*))+$/;
    const arrayPropertyPattern = /(\[])+$/;
 
    return buildConfig(values, configNamePattern, (name, fullName, hasMore, target) => {
        const isArray = arrayPropertyPattern.test(name);
        if (isArray) {
            let arrayDepth = name.match(/\[]/g).length;
            name = name.replace(arrayPropertyPattern, '');
            let array: any[] = target[name] = target[name] || [];
            while (--arrayDepth) {
                array.push(array = []);
            }
            array.push(hasMore ? target = {} : values[fullName]);
        } else {
            target = target[name] = target[name] || (hasMore ? {} : values[fullName]);
        }
        return target;
    });
};
 
const buildMergeConfig = (values: JsonMap): any => {
    const configNamePattern = /^(?=.*\[\d+])CONFIG(:([^:\[\]]+((\[\d+])+(\[])*)?))+$/;
    const arrayPropertyPattern = /(\[\d*])+$/;
 
    return buildConfig(values, configNamePattern, (name, fullName, hasMore, target) => {
        const isArray = arrayPropertyPattern.test(name);
        if (isArray) {
            const propBase = name.replace(arrayPropertyPattern, '');
            let array: any[];
            const arrayIndPattern = /\[(\d*)]/g;
            let arrayIndMatch;
            while ((arrayIndMatch = arrayIndPattern.exec(name)) !== null) {
                const arrayLength = +arrayIndMatch[1] || 0;
                if (array) {
                    array.push(array = Array(arrayLength));
                } else {
                    array = target[propBase] = target[propBase] || Array(arrayLength);
                }
            }
            array.push(hasMore ? target = {} : values[fullName]);
        } else {
            target = target[name] = target[name] || (hasMore ? {} : values[fullName]);
        }
        return target;
    });
};
 
const buildConcatConfigFromEnvVariables = (): any => {
    if ('CONFIG' in process.env) {
        return parseConfig({
            content: process.env.CONFIG,
            description: 'Environment variable config'
        });
    }
    return buildConcatConfig(process.env);
};
 
const buildMergeConfigFromEnvVariables = (): any => {
    if ('CONFIG' in process.env) {
        return {}; // use concat config instead
    }
    return buildMergeConfig(process.env);
};
 
const buildConcatConfigFromOverride = (args: BuildArgs): any => {
    if (args.CONFIG) {
        return parseConfig({
            content: args.CONFIG,
            description: 'Override config'
        });
    }
    return buildConcatConfig(args);
};
 
const buildMergeConfigFromOverride = (args: BuildArgs): any => {
    if (args.CONFIG) {
        return {}; // use concat config instead
    }
    return buildMergeConfig(args);
};
 
const mergeConcat = getMerger({
    inPlace: false,
    arrays: 'concat'
});
 
const mergeConcatPreserve = getMerger({
    inPlace: false,
    arrays: 'concat',
    preserveTypes: true
});
 
const merge = getMerger({
    inPlace: false,
    arrays: 'merge',
    preserveTypes: true
});
 
export const build = async (args: BuildArgs | any): Promise<any> => {
    const [baseConfig, envConfig] = await Promise.all([
        readJson(getBaseConfigFile(args)),
        args.env ? readJson(getEnvConfigFile(args)) : Promise.resolve({})
    ]);
 
    return merge(
        mergeConcatPreserve(
            mergeConcat(
                baseConfig,
                envConfig,
            ),
            buildConcatConfigFromEnvVariables(),
            buildConcatConfigFromOverride(args)
        ),
        buildMergeConfigFromEnvVariables(),
        buildMergeConfigFromOverride(args)
    );
};
